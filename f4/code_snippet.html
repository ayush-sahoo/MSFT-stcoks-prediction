from statsmodels.tsa.stattools import adfuller

def ADF_test(data):
    ST = adfuller(data, autolag = 'AIC')
    print('ADF: ', ST[0])
    print('P-value: ', ST[1])
    print('No. of Lags: ', ST[2])
    print('No. of observations used: ', ST[3])
    print('Critical Values: ')
    for item, value in ST[4].items():
        print('\t', item, ': ', value)
    
    return ST[1]
